<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Scrap-it Tool</title>

    <script src="https://cdn.jsdelivr.net/npm/html2canvas@1.4.1/dist/html2canvas.min.js"></script>

    <script src="https://unpkg.com/dropbox/dist/Dropbox-sdk.min.js"></script>

</head>

<body>

    <button id="scrapItButton" style="position:fixed; right:20px; top:50%; transform:translateY(-50%); z-index:1000;">

        Scrap-it

    </button>


    <script>

        document.getElementById('scrapItButton').addEventListener('click', function() {

            html2canvas(document.body).then(function(canvas) {

                canvas.toBlob(function(blob) {

                    var dbx = new Dropbox.Dropbox({ accessToken: 'YOUR_ACCESS_TOKEN' });

                    var folderName = getUserFolder();

                    if (!folderName) return;


                    var fileName = folderName + '/' + new Date().toISOString() + '.png';

                    dbx.filesUpload({ path: '/' + fileName, contents: blob })

                        .then(function(response) {

                            alert('Screenshot uploaded successfully!');

                        })

                        .catch(function(error) {

                            console.error(error);

                            alert('Error uploading screenshot.');

                        });

                });

            });

        });


        function getUserFolder() {

            let folderName = localStorage.getItem('userFolderName');

            if (!folderName) {

                folderName = prompt('Please enter your folder name:');

                if (folderName) {

                    localStorage.setItem('userFolderName', folderName);

                }

            }

            return folderName;

        }

    </script>

</body>

</html>
